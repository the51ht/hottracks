swiper 공통 가이드
  • 기존에는 new Swiper로 생성하지만 교보 3과제 프로젝트에서는 new CustomSwiper();를 사용하여 스와이퍼를 생성한다.
    - e.g. var swiper = new CustomSwiper();
  • 현재 swiper 가이드에 있는 swiper들은 샘플 코드로 클래스명과 스타일은 상황에 맞춰 바꿔서 사용한다.

swiper 자동롤링

<div class="auto_rolling_test">
	<div class="swiper-container">
		<ul class="swiper-wrapper">
			<li class="swiper-slide"></li>
			...
		</ul>
	</div>
	<div class="swiper_control_box">
		<button type="button" class="btn_control">
			<span class="ico_swiper_control"><span class="hidden">stop</span></span>
		</button>
	</div>
</div>

<script>
	// swiper 재생/멈춤 버튼 타입 script
	var eventBannerWrap = $('.auto_rolling_test');
	var eventSwiper;
	if( $('.swiper-container .swiper-slide', eventBannerWrap).length > 1 ) {
		$('.swiper_control_box', eventBannerWrap).addClass('is_active');
		eventSwiper = new CustomSwiper('.auto_rolling_test .swiper-container', {
			init: false,
			slidesPerView: 1,
			speed: 800,
			observer: true,
			observeParents: true,
			initialSlide: 0,
			autoplay: {
				delay: 1000,
			},
			loop: true,
			on: {
				touchEnd: function () {
					var swiper = this;
					swiper.autoplay.stop();
					swiper.$el.find('.btn_control').addClass('is_stop');
					swiper.$el.find('.btn_control .hidden').html('play');
				}
			}
		});
		eventSwiper.init();

		// 이벤트 베너 - play, stop control 버튼 클릭 event
		$('.swiper_control_box .btn_control').on('click', function () {
			if ($(this).hasClass('is_stop')) {
				// console.log('play');
				$(this).removeClass('is_stop');
				$(this).find('.hidden').html('stop');
				eventSwiper.autoplay.start();
			} else {
				// console.log('stop');
				$(this).addClass('is_stop');
				$(this).find('.hidden').html('play');
				eventSwiper.autoplay.stop();
			}
		});
	}
</script>
